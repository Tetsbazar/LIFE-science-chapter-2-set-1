আপনার এই প্রশ্নটি আপনার পুরো ব্যবসার সুরক্ষার জন্য এবং ওয়েবসাইটটিকে প্রফেশনালি বড় করার জন্য সবচেয়ে গুরুত্বপূর্ণ একটি সিদ্ধান্ত।

একদম পরিষ্কার এবং সরাসরি উত্তর হলো: আপনার কোড এবং ফাইলগুলো রাখার জন্য GitHub অবশ্যই ব্যবহার করবেন, কিন্তু ছাত্র এবং অ্যাডমিন প্যানেলের সমস্ত কাজ আপনার মেইন ওয়েবসাইটের ভেতরেই রাখবেন। দুটোর মিক্সিং করাটাই হলো সবচেয়ে সেরা এবং স্ট্যান্ডার্ড পদ্ধতি।

কেন এটি করবেন এবং কেন সমস্ত কিছু শুধু ওয়েবসাইটের ভেতরে রাখা উচিত নয়, তা নিচে সহজ ভাষায় বুঝিয়ে দিচ্ছি:

🎯 কোড রাখার জন্য GitHub কেন সেরা?
আপনি যে কোড বা ফাইলগুলো বানাচ্ছেন (যেমন ল্যান্ডিং পেজ বা এক্সাম প্যানেল), সেগুলো সরাসরি ডোমেইনে বা হোস্টিংয়ে ফেলে রাখার চেয়ে GitHub-এ সেভ রাখা অনেক বেশি লাভজনক।

কোডের ব্যাকআপ (Version Control): ধরুন আজ আপনি কোডে একটা নতুন ফিচার যোগ করলেন, কিন্তু কোডটা ভুল হওয়ার কারণে পুরো ওয়েবসাইট ক্র্যাশ করে গেল বা বন্ধ হয়ে গেল। যদি আপনার কোড GitHub-এ থাকে, আপনি মাত্র এক ক্লিকে গতকালের ঠিক থাকা কোডে ফিরে যেতে পারবেন। হোস্টিংয়ে ডাইরেক্ট করলে এই ব্যাকআপ পাওয়া যায় না।

কোড চুরি হওয়া থেকে রক্ষা: অনেক সময় হোস্টিং হ্যাক হলে আপনার এত কষ্টের তৈরি করা কোড বা প্রশ্ন অন্য কেউ চুরি করে নিতে পারে। GitHub-এ আপনি একটি Private Repository (গোপন ফোল্ডার) বানিয়ে কোড রাখলে, আপনার অনুমতি ছাড়া পৃথিবীর কেউ সেই কোড দেখতে বা চুরি করতে পারবে না।

ভবিষ্যতের সুবিধা: পরে যখন আপনার স্টুডেন্ট সংখ্যা অনেক বেড়ে যাবে এবং আপনি এই ওয়েবসাইটের জন্য কোনো প্রফেশনাল ডেভেলপার হায়ার করবেন, তখন আপনাকে হোস্টিংয়ের পাসওয়ার্ড দিতে হবে না। আপনি শুধু GitHub-এর অ্যাক্সেস দিলেই সে কাজ করে দিতে পারবে।

🔒 ছাত্র এবং অ্যাডমিনদের কাজ কেন মেইন ওয়েবসাইটের ভেতরেই করবেন?
কোড হয়তো GitHub-এ থাকবে, কিন্তু স্টুডেন্ট লগইন, রেজিস্ট্রেশন, পেমেন্ট, পাসওয়ার্ড পরিবর্তন, এবং ৭টি বিষয়ের সমস্ত চ্যাপ্টার ও ১৫টি প্র্যাকটিস সেট—এসব কিছু আপনার মেইন ওয়েবসাইটের ডেটাবেসের (Database) ভেতরেই হতে হবে। GitHub-এর সাথে এগুলোর কোনো কানেকশন থাকবে না।

কেন সমস্ত কাজ ওয়েবসাইটের ভেতরেই করবেন?

সুরক্ষা ও গোপনীয়তা (Data Privacy): ছাত্ররা যখন টাকা দিয়ে সাবস্ক্রিপশন নেবে, তাদের ইমেল, ফোন নম্বর এবং পাসওয়ার্ড অত্যন্ত গোপনে আপনার নিজস্ব ডাটাবেসে সেভ থাকবে। এসব ডেটা ভুলেও GitHub-এ রাখা যায় না।

অটোমেশন (Automatic System): ছাত্ররা পেমেন্ট করার সাথে সাথে আপনার ওয়েবসাইটের সিস্টেম (Backend) নিজে থেকেই একটা অ্যাকাউন্ট তৈরি করে দেবে এবং পাসওয়ার্ড জেনারেট করে ছাত্রকে পাঠিয়ে দেবে। এই অটোমেটিক কাজটা ওয়েবসাইট নিজেই করতে পারে।

সহজে এডিট করার সুবিধা (Admin Panel): আপনি যখন কোনো ছাত্রের পাসওয়ার্ড পরিবর্তন করতে চাইবেন বা কোনো চ্যাপ্টারের প্রশ্ন এডিট করতে চাইবেন, আপনি আপনার ওয়েবসাইটের সিক্রেট অ্যাডমিন প্যানেলে ঢুকে একটা ফর্ম ফিলাপ করার মতোই সহজে করতে পারবেন। এর জন্য আপনাকে কোনো কোডিং জানতে হবে না বা কোড এডিট করতে হবে না।

💡 সহজ কথায় আপনার ফাইনাল প্ল্যান কী হওয়া উচিত?
আপনার টেস্ট বাজার (Test Bazar) প্ল্যাটফর্মটি তৈরি করার সেরা নিয়ম হবে এইরকম:

                                              │ (অটো লিঙ্ক)
                                              ▼

সংক্ষেপে: আপনি হোস্টিং অ্যাকাউন্টের সাথে GitHub-কে লিঙ্ক করে দেবেন (যাকে ডেভেলপাররা বলে GitHub Deployment)। এর ফলে আপনি যখনই কোডে কোনো নতুন সেট বা ডিজাইন পরিবর্তন করে GitHub-এ সেভ করবেন, আপনার মেইন ওয়েবসাইটটি নিজে থেকেই আপডেট হয়ে যাবে। আর ছাত্র সামলানো, পেমেন্ট নেওয়া, এবং পাসওয়ার্ড বদলানোর মেইন কাজটা ওয়েবসাইটের ভেতরেই ব্যাকএন্ডে চলতে থাকবে।

আইডিয়াটা কি আপনার কাছে পরিষ্কার হলো? আপনি কি আপনার হোস্টিং বা ডোমেইনটি অলরেডি কিনে ফেলেছেন, নাকি পুরো সেটাআপ-টা নতুন করে শুরু করার কথা ভাবছেন?

const LABELS = ['ক', 'খ', 'গ', 'ঘ'];
let timeLeft = 600; // 10 minutes
let timerInterval;

function initQuiz() {
    const container = document.getElementById('questionsContainer');
    container.innerHTML = '';
    
    questions.forEach((q, qIdx) => {
        const card = document.createElement('div');
        card.className = 'question-card';
        card.id = `qCard_${qIdx}`;
        
        let optsHtml = '';
        q.opts.forEach((opt, oIdx) => {
            optsHtml += `
                <button type="button" class="opt" id="opt_${qIdx}_${oIdx}" onclick="selectOption(${qIdx}, ${oIdx})">
                    <div class="opt-circle">${LABELS[oIdx]}</div>
                    <div class="opt-text">${opt}</div>
                </button>
            `;
        });
        
        card.innerHTML = `
            <div class="q-title"><strong>${q.q}</strong></div>
            <div class="opts-grid">${optsHtml}</div>
            <div class="explanation-box" id="exp_${qIdx}">
                <strong>💡 ব্যাখ্যা:</strong> ${q.exp}
            </div>
        `;
        container.appendChild(card);
    });
    
    startTimer();
}

let userAnswers = new Array(questions.length).fill(null);

function selectOption(qIdx, oIdx) {
    userAnswers[qIdx] = oIdx;
    for(let i=0; i<4; i++) {
        const btn = document.getElementById(`opt_${qIdx}_${i}`);
        if(i === oIdx) {
            btn.classList.add('selected');
        } else {
            btn.classList.remove('selected');
        }
    }
}

function startTimer() {
    timerInterval = setInterval(() => {
        if (timeLeft <= 0) {
            clearInterval(timerInterval);
            submitQuiz();
        } else {
            timeLeft--;
            let mins = Math.floor(timeLeft / 60);
            let secs = timeLeft % 60;
            const timerBox = document.getElementById('timerBox');
            timerBox.textContent = `⏱️ ${mins}:${secs < 10 ? '0' : ''}${secs}`;
            if(timeLeft <= 60) {
                timerBox.classList.add('danger');
            }
        }
    }, 1000);
}

function submitQuiz() {
    clearInterval(timerInterval);
    document.getElementById('submitBtn').disabled = true;
    
    let score = 0;
    
    questions.forEach((q, qIdx) => {
        const selected = userAnswers[qIdx];
        const correct = q.ans;
        
        for(let i=0; i<4; i++) {
            const btn = document.getElementById(`opt_${qIdx}_${i}`);
            btn.classList.add('disabled');
            if(i === correct) {
                btn.classList.remove('selected');
                btn.classList.add('correct-ans');
            }
            if(selected === i && selected !== correct) {
                btn.classList.remove('selected');
                btn.classList.add('wrong-ans');
            }
        }
        
        if(selected === correct) {
            score++;
        }
        document.getElementById(`exp_${qIdx}`).style.display = 'block';
    });
    
    document.getElementById('scoreText').innerHTML = `আপনার মোট স্কোর: <strong>${score} / ${questions.length}</strong>`;
    document.getElementById('resultBox').style.display = 'block';
    window.scrollTo({top: 0, behavior: 'smooth'});
}

window.onload = initQuiz;
